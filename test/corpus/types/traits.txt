trait Functor {
  map: (Self<a>, (a) -> b) -> Self<b>
}

impl Functor for DynamicArray<t> {
  map = (arr, fn) => {
    let result: []t = []
    for idx, item in arr {
      result.push(fn(item))
    }
    result
  }
}

impl Functor for Maybe {
  map = (maybe, fn) => match maybe {
    Some x => Some fn(x),
    None => None
  }
}

trait Eq<t> {
  (_==_): (t, t) -> bool,
  (_!=_): (t, t) -> bool
}

impl Eq<t> for []t {
  (_==_) = (a, b) => {
    if a.length() != b.length() {
      false
    } else {
      for i in 0..<a.length() {
        if a[i] != b[i] {
          return false
        }
      }
      true
    }
  },
  (_!=_) = (a, b) => !(a == b)
}

trait Collection<c, e> {
  add: (c<e>, e) -> c<e>,
  remove: (c<e>, e) -> c<e>,
  contains: (c<e>, e) -> bool,
  size: (c<e>) -> int
}

impl Collection<Array, string> for StringList {
  add = (arr, item) => [...arr, item],
  remove = (arr, item) => filter(arr, (x) => x != item),
  contains = (arr, item) => any(arr, (x) => x == item),
  size = (arr) => arr.length()
}
